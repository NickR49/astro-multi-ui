From within an Astro file, components from different frameworks can be nested within each other -

<br />
<ReactCard client:load>
  <SolidCard client:load>

  </SolidCard>
  <SvelteCard client:load>

  </SvelteCard>
</ReactCard>
<SolidCard client:load>
  <ReactCard client:load>

      <SvelteCard client:load />
    </VueCard>
  </ReactCard>
</SolidCard>
